First, create a handler to list all stored content:

{`import { Storage } from '@vudo/runtime';
import { Scene, Passage } from '../ontology/main.dol';

interface GalleryItem {
  id: string;
  timestamp: number;
  scene: Scene;
  imagePath: string;
  passage?: Passage;
  thumbnailPath?: string;
}

interface GalleryData {
  items: GalleryItem[];
  totalCount: number;
  lastUpdated: Date;
}

export async function loadGalleryData(): Promise<GalleryData> {
  // List all metadata files
  const metadataFiles = await Storage.list('output/metadata/*.json');

  const items: GalleryItem[] = [];

  for (const file of metadataFiles) {
    const metadata = await Storage.readJson(file);
    const id = file.replace('output/metadata/', '').replace('.json', '');

    // Check for corresponding passage
    let passage: Passage | undefined;
    const passageFile = \`output/passages/\${id}.json\`;
    if (await Storage.exists(passageFile)) {
      const passageData = await Storage.readJson(passageFile);
      passage = passageData.passage;
    }

    items.push({
      id,
      timestamp: parseInt(id),
      scene: metadata.scene,
      imagePath: metadata.localPath,
      passage,
      thumbnailPath: metadata.localPath.replace('.png', '_thumb.png')
    });
  }

  // Sort by timestamp, newest first
  items.sort((a, b) => b.timestamp - a.timestamp);

  return {
    items,
    totalCount: items.length,
    lastUpdated: new Date()
  };
}

export async function getGalleryItem(id: string): Promise<GalleryItem | null> {
  const metadataFile = \`output/metadata/\${id}.json\`;

  if (!await Storage.exists(metadataFile)) {
    return null;
  }

  const metadata = await Storage.readJson(metadataFile);

  let passage: Passage | undefined;
  const passageFile = \`output/passages/\${id}.json\`;
  if (await Storage.exists(passageFile)) {
    const passageData = await Storage.readJson(passageFile);
    passage = passageData.passage;
  }

  return {
    id,
    timestamp: parseInt(id),
    scene: metadata.scene,
    imagePath: metadata.localPath,
    passage
  };
}`}
  </pre>
</div>

{`import { Storage, View } from '@vudo/runtime';
import { loadGalleryData } from './gallery';

export async function main(): Promise<void> {
  // Load the HTML template
  const template = await Storage.readText('src/views/gallery.html');

  // Load gallery data
  const data = await loadGalleryData();

  // Inject data into the template
  const html = template.replace(
    '/* GALLERY_DATA_PLACEHOLDER */',
    JSON.stringify(data)
  );

  // Serve the view
  await View.serve(html, {
    title: 'Lion & Swan Gallery',
    contentType: 'text/html'
  });
}`}
